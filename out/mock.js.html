import mock from 'mockjs';

/**
 * 公共请求返回配置
 * @param {Number} errcode 接口请求状态
 */
const schema = (errcode = 0) => {
    let returnData = null;
    errcode === 0 ? returnData = {
        'errcode' : 0,
        'errmsg' : '成功',
    } : returnData = {
        'errcode' : 1,
        'errmsg' : '失败',
    };
    return returnData;
}
